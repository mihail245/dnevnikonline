Настройка Firebase

  1. Перейдите на консоль Firebase
  2. Создайте новый проект или выберите существующий
  3. В разделе "Authentication" включите вход по email/password
  4. В разделе "Firestore Database" создайте базу данных в режиме тестовой среды
  5. Установите следующие правила безопасности для Firestore:
rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    match /users/{userId} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
      
      match /subjects/{subject} {
        allow read, write: if request.auth != null && request.auth.uid == userId;
      }
    }
  }
}
